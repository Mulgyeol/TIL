# Destructuring Assignment

  • ref (opens new window)
{
  // object => {}
  const student = {
    name: "Anna",
    level: 1,
  };

  // bad
  {
    const name = student.name;
    const level = student.level;
    console.log(name, level);
  }

  // bad

    const first = animals[0];
